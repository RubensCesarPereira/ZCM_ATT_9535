method ARQUIVOSET_GET_ENTITYSET.
  DATA: ls_it2001 TYPE zhrtit2001,
        lt_it2001 TYPE TABLE OF zhrtit2001,

        ls_entity      LIKE LINE OF et_entityset,

        lv_pernr       TYPE pa0105-pernr,
        lv_nome        TYPE string.

  data: s_subty        type /iwbep/t_cod_select_options.

  field-symbols <fs_filter> type /iwbep/s_mgw_select_option.

    LOOP AT it_filter_select_options ASSIGNING <fs_filter>.
      CASE <fs_filter>-property.
        WHEN 'subty'.
          s_subty = <fs_filter>-select_options.
      ENDCASE.
    ENDLOOP.

    SELECT SINGLE pernr
      FROM pa0105
      INTO lv_pernr
      WHERE subty EQ '0001'
      AND   usrid EQ sy-uname
      AND   endda EQ '99991231'.

    CHECK lv_pernr IS NOT INITIAL.

    SELECT * FROM zhrtit2001 INTO TABLE lt_it2001
    WHERE pernr EQ lv_pernr
      AND subty IN s_subty.

    LOOP AT lt_it2001 INTO  ls_it2001.

      ls_entity-nome = ls_it2001-nome_arq. "Nome do arquivo
      ls_entity-documentid = ls_it2001-nome_arq. "Nome do arquivo
      ls_entity-filename = ls_it2001-nome_arq. "Nome do arquivo
      ls_entity-subty = ls_it2001-subty. "Subty

      APPEND ls_entity TO et_entityset.

    ENDLOOP.
endmethod.

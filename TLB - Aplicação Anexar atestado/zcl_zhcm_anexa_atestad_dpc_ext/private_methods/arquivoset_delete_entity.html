METHOD arquivoset_delete_entity.
  DATA: lv_message TYPE bapiret2-message.

  DATA: lv_nome TYPE string,
        lv_pernr TYPE pa0105-pernr,
        lv_subty TYPE pa0021-subty.
*        lv_message TYPE string.

  DATA: ls_zhrtit2001 TYPE zhrtit2001.
  FIELD-SYMBOLS <fs_key> TYPE /iwbep/s_mgw_name_value_pair.

  LOOP AT it_key_tab ASSIGNING <fs_key>.

    ENDCASE.
  ENDLOOP.

  SELECT SINGLE pernr
    FROM pa0105
    INTO lv_pernr
    WHERE subty EQ '0001'
    AND   usrid EQ sy-uname
    AND   endda EQ '99991231'.
  IF sy-subrc EQ 0.
    SELECT SINGLE *
      FROM zhrtit2001
      INTO ls_zhrtit2001
      WHERE pernr EQ lv_pernr
      AND   nome_arq EQ lv_nome
      AND   subty    EQ lv_subty.
    IF sy-subrc EQ 0.
      DELETE FROM zhrtit2001 WHERE pernr = lv_pernr
                               AND nome_arq = lv_nome
                               AND subty    = lv_subty.
      IF sy-subrc NE 0.
        lv_message = 'Erro ao excluir o registro'..
        RAISE EXCEPTION TYPE /iwbep/cx_mgw_busi_exception
          EXPORTING
            textid  = /iwbep/cx_mgw_busi_exception=>business_error
            message = lv_message.
      ENDIF.
    ENDIF.
  ENDIF.
ENDMETHOD.

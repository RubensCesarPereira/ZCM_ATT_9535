METHOD buscacidset_get_entityset.
  DATA: lt_t7ehs00_diagslt TYPE TABLE OF t7ehs00_diagslt,
        ls_t7ehs00_diagslt TYPE t7ehs00_diagslt,
        ls_entity LIKE LINE OF et_entityset.

  SELECT * FROM
    t7ehs00_diagslt
    INTO TABLE lt_t7ehs00_diagslt
    WHERE diagn_lang EQ 'PT'.
  IF sy-subrc EQ 0.
    LOOP AT lt_t7ehs00_diagslt INTO ls_t7ehs00_diagslt.
      ls_entity-diagn_code = ls_t7ehs00_diagslt-diagn_code.
      ls_entity-diagn_mc = ls_t7ehs00_diagslt-diagn_mc.

      APPEND ls_entity TO et_entityset.
*      SORT et_entityset BY diagn_code ASCENDING.
    ENDLOOP.
  ENDIF.
ENDMETHOD.

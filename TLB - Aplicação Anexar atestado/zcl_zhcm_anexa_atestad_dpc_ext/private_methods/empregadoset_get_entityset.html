METHOD empregadoset_get_entityset.
  DATA: ls_pa0001      TYPE pa0001,
        lv_pernr       TYPE pa0105-pernr,
        lv_email       TYPE pa0105-usrid_long,
        lv_cname       TYPE pa0002-cname,
        lv_comercial   TYPE pa0006-num01,
        lv_celular     TYPE pa0006-num02,
        ls_t513s       TYPE t513s,
        ls_t500p       TYPE t500p,
        ls_t001p       TYPE t001p,
        ls_entity LIKE LINE OF et_entityset.

  SELECT SINGLE pernr
    FROM pa0105
    INTO lv_pernr
    WHERE subty EQ '0001'
    AND   usrid EQ sy-uname
    AND   endda EQ '99991231'.

  CHECK lv_pernr IS NOT INITIAL.

  SELECT SINGLE cname
    FROM pa0002
    INTO lv_cname
    WHERE pernr EQ lv_pernr.

  SELECT SINGLE num01
    FROM pa0006
    INTO lv_comercial
    WHERE pernr EQ lv_pernr
    AND   com01 EQ 'COM'.

  SELECT SINGLE num02
    FROM pa0006
    INTO lv_celular
    WHERE pernr EQ lv_pernr
    AND   com02 EQ 'CEL'.

  SELECT SINGLE usrid_long
    FROM pa0105
    INTO lv_email
    WHERE pernr EQ lv_pernr
    AND   subty EQ '0010'
    AND   endda EQ '99991231'.

  ls_entity-pernr = lv_pernr.
  ls_entity-cname = lv_cname.
  ls_entity-email = lv_email.
  ls_entity-tel   = lv_comercial.
  ls_entity-cel   = lv_celular.

  SELECT SINGLE *
    FROM pa0001
    INTO ls_pa0001
    WHERE pernr EQ lv_pernr.
  IF sy-subrc EQ 0.

    SELECT SINGLE *
      FROM t513s
      INTO ls_t513s
      WHERE sprsl EQ 'PT'
      AND   stell EQ ls_pa0001-stell
      AND   endda EQ '99991231'.
    IF sy-subrc EQ 0.
      ls_entity-cargo = ls_t513s-stltx.
    ENDIF.

    SELECT SINGLE *
      FROM t500p
      INTO ls_t500p
      WHERE persa EQ ls_pa0001-werks.
    IF sy-subrc EQ 0.
      ls_entity-area = ls_t500p-name1.
    ENDIF.

    SELECT SINGLE *
      FROM t001p
      INTO ls_t001p
      WHERE werks EQ ls_pa0001-werks
      AND   btrtl EQ ls_pa0001-btrtl.
    IF sy-subrc EQ 0.
      ls_entity-subarea = ls_t001p-btext.
    ENDIF.
  ENDIF.

  APPEND ls_entity TO et_entityset.
ENDMETHOD.
